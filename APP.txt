        ' Fonction d'echappement HTML pour prevenir les failles XSS
        Function VBS_EscapeHTML(ByVal str)
            If IsNull(str) Or IsEmpty(str) Then
                VBS_EscapeHTML = ""
                Exit Function
            End If
            Dim res
            res = CStr(str)
            res = Replace(res, "&", "&amp;")
            res = Replace(res, "<", "&lt;")
            res = Replace(res, ">", "&gt;")
            res = Replace(res, """", "&quot;")
            res = Replace(res, "'", "&#39;")
            VBS_EscapeHTML = res
        End Function

        Sub WriteLog(msg)
            On Error Resume Next
            Dim fso, logStream, logPath, wsh
            Set wsh = CreateObject("WScript.Shell")
            logPath = wsh.ExpandEnvironmentStrings("%APPDATA%") & "\SAP_Cloture_Log.txt"
            Set fso = CreateObject("Scripting.FileSystemObject")
            Set logStream = fso.OpenTextFile(logPath, 8, True) 
            logStream.WriteLine Now & " | " & msg
            logStream.Close
            Set logStream = Nothing
            Set fso = Nothing
            Set wsh = Nothing
            On Error GoTo 0
        End Sub

        Function GetSAPSession()
            On Error Resume Next
            Set GetSAPSession = Nothing
            Dim SapGuiAuto, appSAP, conn
            Set SapGuiAuto = GetObject("SAPGUI")
            If Err.Number <> 0 Then Exit Function
            Set appSAP = SapGuiAuto.GetScriptingEngine
            If appSAP.Children.Count = 0 Then Exit Function
            Set conn = appSAP.Children(0)
            If conn.Children.Count = 0 Then Exit Function
            Set GetSAPSession = conn.Children(0)
            On Error GoTo 0
        End Function

        Sub PauseVBS(secondes)
            On Error Resume Next
            Dim wshShell
            Set wshShell = CreateObject("WScript.Shell")
            wshShell.Run "cmd /c ping localhost -n " & (secondes + 1) & " > nul", 0, True
            Set wshShell = Nothing
            On Error GoTo 0
        End Sub

        Function AttendreElementSAP(cheminID, maxSecondes)
            Dim tentatives, obj
            tentatives = 0
            AttendreElementSAP = False
            On Error Resume Next
            Do While tentatives < maxSecondes
                Set obj = oSession.FindById(cheminID, False)
                If Not obj Is Nothing Then
                    AttendreElementSAP = True
                    On Error GoTo 0
                    Exit Function
                End If
                Call PauseVBS(1)
                tentatives = tentatives + 1
            Loop
            On Error GoTo 0
        End Function

        Function OuvrirQM02_Dynamique(icNum)
            Dim msgType, msgText, tentatives
            tentatives = 0
            Do While tentatives < 3
                oSession.EndTransaction
                oSession.StartTransaction "QM02"
                If AttendreElementSAP("wnd[0]/usr/ctxtRIWO00-QMNUM", 5) Then
                    oSession.FindById("wnd[0]/usr/ctxtRIWO00-QMNUM").Text = icNum

                Else
                    OuvrirQM02_Dynamique = "ERREUR: SAP trop lent (Timeout QM02)"
                    Exit Function
                End If
                
                On Error Resume Next
                msgType = ""
                msgText = ""
                msgType = oSession.FindById("wnd[0]/sbar").MessageType
                msgText = oSession.FindById("wnd[0]/sbar").Text

                Else

        Function VBS_CreerICs(strDate, strVacation, strEquipe, strData, strOP, strIQ, strST)
            Dim arrRows, i, rowData, arrFields, y, r, msn, numOF, fullDesc, zoneI, zoneA, chkBox, strPoste
            Dim resultString, createdIC, reponseMsgBox, Texte, DescPart, Titre, H1, H2, PosteSAP, DistribSAP
            Dim arrOP, arrIQ, arrST, countOP, countIQ, countST, NB_INTER, rIndex, rObj, rText, dateSapFormat
            Dim isToday, currentHour, currentMin, totalMins, futureTime, hPlus2
            
            On Error Resume Next
            arrOP = Split(strOP, ";")
            arrIQ = Split(strIQ, ";")
            arrST = Split(strST, ";")
            
            countOP = 0
            countIQ = 0
            countST = 0
            
            If strOP <> "" Then countOP = UBound(arrOP) + 1
            If strIQ <> "" Then countIQ = UBound(arrIQ) + 1
            If strST <> "" Then countST = UBound(arrST) + 1
            
            arrRows = Split(strData, "§")
            resultString = ""
    
            Set oSession = GetSAPSession()
            If oSession Is Nothing Then 
                VBS_CreerICs = "ERREUR : SAP fermé."
                Exit Function
            End If
         
            dateSapFormat = Replace(strDate, "/", ".")
            
            For i = 0 To UBound(arrRows)
                rowData = arrRows(i)
                arrFields = Split(rowData, "|")
               
                If UBound(arrFields) >= 6 Then
                    msn = arrFields(0)
                    numOF = arrFields(1)
                    fullDesc = arrFields(2)
                    zoneI = arrFields(3)
                    zoneA = arrFields(4)
                    chkBox = arrFields(5)
                    strPoste = UCase(arrFields(6))
                    
                    If InStr(1, fullDesc, "//") = 0 Then 
                        VBS_CreerICs = "ERREUR : Format descriptif."
                        Exit Function
                    End If
                   
                    Texte = Trim(Left(fullDesc, InStr(1, fullDesc, "//") - 1))
                    DescPart = Trim(Right(fullDesc, Len(fullDesc) - InStr(1, fullDesc, "//") - 1))
                    Titre = numOF & " - " & Left(DescPart, 25)
              
                    isToday = False
                    If strDate = (Right("0" & Day(Date), 2) & "/" & Right("0" & Month(Date), 2) & "/" & Year(Date)) Then
                        isToday = True
                    End If
                    
                    currentHour = Hour(Now)
                    currentMin = Minute(Now)
                    totalMins = (currentHour * 60) + currentMin
                    
                    If strVacation = "Jour" Then 
                        H1 = "07:00"
                        H2 = "14:45" 
                    Else 
                        H1 = "14:45"
                        H2 = "23:59"
                    End If
     
                    If zoneI = "CC11" Or chkBox = "P-OFF" Then
                        If strVacation = "Jour" Then 
                            H1 = "11:00"
                            H2 = "12:00" 
                            If isToday And totalMins >= (11 * 60) Then
                                MsgBox "L'horaire de création pour CC11 / P-OFF (Vacation Jour 11h-12h) est dépassé pour la journée en cours." & vbCrLf & "L'intervention sur le MSN " & msn & " est annulée.", 48, "Horaire dépassé"
                                VBS_CreerICs = "Annulé"
                                Exit Function
                            End If
                        Else 
                            H1 = "19:00"
                            H2 = "20:00"
                            If isToday And totalMins >= (19 * 60) Then
                                MsgBox "L'horaire de création pour CC11 / P-OFF (Vacation Soir 19h-20h) est dépassé pour la journée en cours." & vbCrLf & "L'intervention sur le MSN " & msn & " est annulée.", 48, "Horaire dépassé"

                        End If
                    Else
                        If isToday Then
                            futureTime = DateAdd("n", 2, Now)
                            hPlus2 = Right("0" & Hour(futureTime), 2) & ":" & Right("0" & Minute(futureTime), 2)
                            
                            If strVacation = "Jour" Then
                                If totalMins >= (7 * 60) And totalMins < (14 * 60 + 43) Then
                                    H1 = hPlus2
                                ElseIf totalMins >= (14 * 60 + 43) Then

                            End If

                    End If
                    
                    PosteSAP = ""

        function FormatDateString(dateObj) {
            var dd = dateObj.getDate();
            var mm = dateObj.getMonth() + 1; var yyyy = dateObj.getFullYear();
            if (dd < 10) dd = '0' + dd;
            if (mm < 10) mm = '0' + mm;
            return dd + '/' + mm + '/' + yyyy;
        }

        function parseDateString(dateStr) {
            if (!dateStr) return null;
            var cleanStr = dateStr.replace(/[\.-]/g, '/'); 
            var parts = cleanStr.split('/'); if (parts.length !== 3) return null;
            var d, m, y;
            if (parts[0].length === 4) {
                y = parseInt(parts[0], 10);
                m = parseInt(parts[1], 10) - 1; d = parseInt(parts[2], 10);
            } else {
                d = parseInt(parts[0], 10);
                m = parseInt(parts[1], 10) - 1;
                y = parts[2].length === 4 ? parseInt(parts[2], 10) : parseInt(parts[2], 10) + 2000;
            }
            if (isNaN(d) || isNaN(m) || isNaN(y)) return null;
            return new Date(y, m, d);
        }

        function buildCalendar(month, year) {
            var dp = document.getElementById('custom-datepicker');
            var html = '<div class="dp-header"><button type="button" class="prev" onclick="changeDpMonth(-1, event)">&lt;</button><span>' + moisNoms[month] + ' ' + year + '</span><button type="button" class="next" onclick="changeDpMonth(1, event)">&gt;</button><div style="clear:both;"></div></div><table class="dp-table"><tr><th>Lu</th><th>Ma</th><th>Me</th><th>Je</th><th>Ve</th><th>Sa</th><th>Di</th></tr><tr>';
            var firstDay = new Date(year, month, 1).getDay(); if (firstDay === 0) firstDay = 7;
            var daysInMonth = new Date(year, month + 1, 0).getDate(); var d = 1;
            for (var i = 1; i < firstDay; i++) { html += '<td></td>'; }
            for (var i = firstDay; i <= 7; i++) { html += '<td onclick="pickDpDate('+d+','+month+','+year+')"><span>'+d+'</span></td>'; d++; }
            html += '</tr>';
            while (d <= daysInMonth) {
                html += '<tr>';
                for (var i = 1; i <= 7; i++) {
                    if (d <= daysInMonth) { html += '<td onclick="pickDpDate('+d+','+month+','+year+')"><span>'+d+'</span></td>'; d++; } else { html += '<td></td>'; }
                }
                html += '</tr>';
            }
            dp.innerHTML = html + '</table>';
        }

        function getNbLignesCreation() {
            var rowH = 39;
            // Si le conteneur est déjà dimensionné (page affichée), on cale au pixel près
            // en mesurant la hauteur réelle d'une ligne et de l'en-tête.
            var container = document.getElementById("containerTableCreation");
            var thead = document.querySelector("#tableSaisieCreation thead");
            var tbody = document.getElementById("tbodySaisieCreation");
            if (container && container.offsetHeight > 0) {
                var theadH = thead ? thead.offsetHeight : 36;
                if (tbody && tbody.rows.length > 0 && tbody.rows[0].offsetHeight > 0) rowH = tbody.rows[0].offsetHeight;
                // Petite marge de 3px (bordures) pour ne jamais déclencher la scroll bar.
                return Math.max(12, Math.floor((container.clientHeight - theadH - 3) / rowH));
            }
            // Fallback au démarrage (page pas encore affichée) : estimation sur la fenêtre.
            var winH = document.documentElement.clientHeight || window.innerHeight || screen.availHeight || 1040;
            return Math.max(12, Math.floor((winH - 300) / rowH));

        function LancerCreationSAP(btnElement) {
            var dateSouhaitee = document.querySelector('#inputDateCreation').value;
            var radVac = document.getElementsByName('radVacation'); var vacation = "";
            for(var r=0; r<radVac.length; r++) { if(radVac[r].checked) vacation = radVac[r].value; }
            var radEq = document.getElementsByName('radEquipe'); var equipe = "";
            for(var r=0; r<radEq.length; r++) { if(radEq[r].checked) equipe = radEq[r].value; }
            if (!dateSouhaitee) { ErreurCreation("Veuillez saisir une Date Souhaitée."); return; }

            // Date : interdiction de créer une IC pour une date antérieure à aujourd'hui.
            var maintenant = new Date();
            var aujourdhui = new Date(maintenant.getFullYear(), maintenant.getMonth(), maintenant.getDate());
            var dateObjCreation = parseDateString(dateSouhaitee);
            if (dateObjCreation) {
                var dateIC = new Date(dateObjCreation.getFullYear(), dateObjCreation.getMonth(), dateObjCreation.getDate());
                if (dateIC < aujourdhui) { ErreurCreation("La date est antérieure à aujourd'hui."); return; }
            }

            // Heure : passé 14h40, la vacation Jour du jour même n'est plus créable.
            var estAujourdhui = dateObjCreation &&
                dateObjCreation.getFullYear() === aujourdhui.getFullYear() &&
                dateObjCreation.getMonth() === aujourdhui.getMonth() &&
                dateObjCreation.getDate() === aujourdhui.getDate();
            if (vacation === "Jour" && estAujourdhui &&
                (maintenant.getHours() > 14 || (maintenant.getHours() === 14 && maintenant.getMinutes() > 40))) {
                ErreurCreation("Il est plus de 14h40 : impossible de créer une IC pour la vacation de Jour.");
                return;
            }

            var arrOP = [], arrIQ = [], arrST = [];
            for (var p = 0; p < personnelData.length; p++) {
                if (personnelData[p].equipe === equipe) {
                    if (personnelData[p].poste === "OP") arrOP.push(personnelData[p].nom);
                    if (personnelData[p].poste === "IQ") arrIQ.push(personnelData[p].nom);
                    if (personnelData[p].poste === "ST") arrST.push(personnelData[p].nom);
                }
            }

            // Effectif : impossible de créer si l'équipe sélectionnée n'a aucun membre.
            if (arrOP.length === 0 && arrIQ.length === 0 && arrST.length === 0) {
                ErreurCreation("L'équipe sélectionnée n'a aucun effectif.");
                return;
            }

            var tbody = document.getElementById("tbodySaisieCreation");
            var validRowsData = [];
            var chkNames = ["QLS", "NDT", "P-OFF", "ST", "VSD", "RLQ1", "RLQ2"];
            for (var i = 0; i < tbody.rows.length; i++) {
                var row = tbody.rows[i];
                var msn = row.cells[1].querySelector('input').value.replace(/^\s+|\s+$/g, "").toUpperCase();
                var numOF = row.cells[2].querySelector('input').value.replace(/^\s+|\s+$/g, "");
                var desc = row.cells[3].querySelector('input').value.replace(/^\s+|\s+$/g, "");
                var zoneI = row.cells[4].querySelector('select').value;
                var zoneA = getZoneAvionFromInterv(zoneI);
                var existingIC = row.cells[12].querySelector('div').innerText;

                // Une ligne est "à créer" dès qu'au moins un champ est renseigné.
                var uneCaseCochee = false;
                for (var cc = 0; cc < 7; cc++) { if (row.cells[5 + cc].querySelector('input').checked) { uneCaseCochee = true; break; } }
                var ligneRemplie = (msn !== "" || numOF !== "" || desc !== "" || zoneI !== "" || uneCaseCochee);

                if (ligneRemplie && (existingIC === "---" || existingIC === "")) {
                    if (msn === "") { ErreurCreation("Ligne " + (i + 1) + " : le MSN est obligatoire."); return; }
                    if (numOF === "") { ErreurCreation("Ligne " + (i + 1) + " : le Numéro OF est obligatoire."); return; }
                    if (desc === "") { ErreurCreation("Ligne " + (i + 1) + " : la Description est obligatoire."); return; }
                    if (zoneI === "") { ErreurCreation("Ligne " + (i + 1) + " : la Zone interv. est obligatoire."); return; }
                    if (!/^[0-9]{8,}$/.test(numOF)) { ErreurCreation("Ligne " + (i + 1) + " : le Numéro OF doit contenir uniquement des chiffres (minimum 8 chiffres)."); return; }
                    if (desc.indexOf("//") === -1) { ErreurCreation("Ligne " + (i + 1) + " : La description doit respecter le format \"Zone// Travail\"."); return; }
                    if (desc.indexOf("|") !== -1 || desc.indexOf("§") !== -1) { ErreurCreation("Ligne " + (i + 1) + " : La description ne doit pas contenir les caractères '|' ou '§'."); return; }
                    if (msn.indexOf("|") !== -1 || msn.indexOf("§") !== -1) { ErreurCreation("Ligne " + (i + 1) + " : Le MSN ne doit pas contenir les caractères '|' ou '§'."); return; }
                    var posteFAL = "";
                    if (msn !== "") {
                        var msnObj = null;
                        for (var m = 0; m < msnData.length; m++) { if (msnData[m].msn === msn) { msnObj = msnData[m]; break; } }
                        if (msnObj) {
                            posteFAL = msnObj.poste;
                            var parsedSortie = parseDateString(msnObj.dateSortie.replace(/^\s+|\s+$/g, ""));
                            if (parsedSortie && parsedSortie < new Date()) { ErreurCreation("Ligne " + (i + 1) + " : le MSN " + msn + " est périmé (date de sortie : " + msnObj.dateSortie.replace(/^\s+|\s+$/g, "") + ")."); return; }
                        } else { ErreurCreation("Ligne " + (i + 1) + " : MSN " + msn + " introuvable."); return; }
                    }
                    var checkedBox = "";
                    for(var c = 0; c < 7; c++) { if(row.cells[5+c].querySelector('input').checked) { checkedBox = chkNames[c]; break; } }
                    validRowsData.push({ index: i, dataStr: msn + "|" + numOF + "|" + desc + "|" + zoneI + "|" + zoneA + "|" + checkedBox + "|" + posteFAL });
                }
            }
            if (validRowsData.length === 0) { ErreurCreation("Aucune ligne valide."); return; }
            stopCreationFlag = false; var oldText = btnElement.innerHTML;
            btnElement.innerHTML = "<div class='spinner'></div>"; btnElement.disabled = true;
            TraiterCreationLigneSuivante(0, validRowsData, dateSouhaitee, vacation, equipe, arrOP.join(";"), arrIQ.join(";"), arrST.join(";"), btnElement, oldText);
        }

        function UpdateValueMSN(index, field, value) { 
            if (field === "msn") {
                msnData[index].msn = value.toUpperCase().replace(/^\s+|\s+$/g, "");

                    msnData[index].poste = ""; msnData[index].dateSortie = ""; msnData[index].valide = false;
                    SaveData(); RenderMSNTable(); return;

            if (field === "poste") msnData[index].poste = value.toUpperCase().replace(/^\s+|\s+$/g, "");
            if (field === "dateSortie") msnData[index].dateSortie = value.replace(/^\s+|\s+$/g, ""); 
            SaveData();
        }

        document.onpaste = function(e) {
            var ev = e || window.event; 
            if (!ev) return true;
            var target = ev.target || ev.srcElement;
            var table = getClosestTag(target, 'TABLE');
            if (!table || table.id !== 'tableSaisieCreation') return true; 

            var pastedText = "";
            if (window.clipboardData && window.clipboardData.getData) {
                pastedText = window.clipboardData.getData('Text');
            } else if (ev.clipboardData && ev.clipboardData.getData) {
                pastedText = ev.clipboardData.getData('text/plain');
            }
            if (!pastedText) return true;

            if(ev.preventDefault) ev.preventDefault();
            else ev.returnValue = false;

            try {
                var rows = pastedText.split('\n');
                var tr = getClosestTag(target, 'TR'); 
                var tbodyInner = tr.parentNode;
                var startRowIndex = -1, startColIndex = -1;
                for(var r=0; r<tbodyInner.children.length; r++) { if(tbodyInner.children[r]===tr) { startRowIndex=r; break; } }
                var td = getClosestTag(target, 'TD');
                for(var c=0; c<tr.children.length; c++) { if(tr.children[c]===td) { startColIndex=c; break; } }

                for (var i = 0; i < rows.length; i++) {
                    var rowStr = rows[i].replace(/\r/g, '');
                    if (rowStr === '' && i === rows.length - 1) continue;
                    if (startRowIndex + i >= tbodyInner.children.length) AjouterLigneSaisieIC();
                    var currentRow = tbodyInner.children[startRowIndex + i]; var cells = rowStr.split('\t');
                    for (var j = 0; j < cells.length; j++) {
                        if (startColIndex + j >= currentRow.children.length) continue;
                        var inputEl = currentRow.children[startColIndex + j].querySelector('input, select');
                        if (inputEl) {
                            var val = NettoyerChaine(cells[j]);
                            if (inputEl.type === 'checkbox') {
                                var vLow = val.toLowerCase();
                                var shouldCheck = (vLow === 'vrai' || vLow === 'true' || vLow === '1' || vLow === 'x' || vLow === 'oui' || vLow === 'yes');
                                inputEl.checked = shouldCheck; if(shouldCheck) uncheckOthers(inputEl);
                            } else if (inputEl.tagName === 'SELECT') {
                                for(var opt = 0; opt < inputEl.options.length; opt++) {
                                    if(inputEl.options[opt].value.toUpperCase() === val.toUpperCase()) { inputEl.selectedIndex = opt; break; }
                                }
                            } else { inputEl.value = val; }
                        }
                    }
                }
            } catch (err) { window.status = "Avertissement : collage partiel — " + err.message; }
            return false;
        };

        // Nombre de lignes qui tiennent dans une carte capée au viewport (corps défilant).
        // Doit être mesuré conteneur VISIBLE, sinon les hauteurs valent 0 et le résultat explose.
        function ComputeEditFitRows() {
            var firstCard = document.querySelector("#viewModePlanningEquipes .card");
            var thead = document.querySelector("#tablePlanningMSN thead");
            if (firstCard && thead && firstCard.offsetHeight > 0) {
                var clientH = document.documentElement.clientHeight || window.innerHeight;
                var cardTop = firstCard.getBoundingClientRect().top;
                var availableCard = clientH - cardTop - 25;          // hauteur max de la carte
                var titleH = firstCard.children[0] ? firstCard.children[0].offsetHeight : 0;
                var bodyContent = availableCard - titleH - 30;        // - padding haut/bas du corps
                var rowsArea = bodyContent - thead.offsetHeight - 20; // - en-tête - spacer bas
                return Math.max(8, Math.floor(rowsArea / 38));
            }
            return Math.max(8, Math.floor((window.innerHeight - 285) / 38));
        }

        function getTotalEditRows() {
            // Valeur mémorisée par ToggleEditMode (mesurée conteneur visible). Repli si absente.
            if (typeof window.editFitRows === "number" && window.editFitRows > 0) return window.editFitRows;
            return ComputeEditFitRows();
        }

        function SaveAndExitEditMode() {
            // 1. Sauvegarder les NOUVEAUX membres (ceux ajoutés tout en bas)
            var newInputs = document.querySelectorAll(".new-member-input");
            for (var i = 0; i < newInputs.length; i++) {
                var nom = newInputs[i].value.replace(/^\s+|\s+$/g, "");
                if (!nom) continue;
                var eqName = newInputs[i].getAttribute("data-eqname");
                var row = newInputs[i].parentNode.parentNode.parentNode;
                var sel = (row && row.cells && row.cells[0]) ? row.cells[0].getElementsByTagName("select")[0] : null;
                var poste = (sel && sel.value) ? sel.value : "OP";
                personnelData.push({nom: nom.toUpperCase(), poste: poste, equipe: eqName});
            }

            // 2. Forcer la sauvegarde des ANCIENS membres modifiés (contourne le bug du clic trop rapide)
            var existingInputs = document.querySelectorAll("input[data-pidx]");
            for(var i = 0; i < existingInputs.length; i++) {
                var idx = parseInt(existingInputs[i].getAttribute("data-pidx"));
                var updatedNom = existingInputs[i].value.toUpperCase().replace(/^\s+|\s+$/g, "");
                
                var row = existingInputs[i].parentNode.parentNode.parentNode;
                var sel = (row && row.cells && row.cells[0]) ? row.cells[0].getElementsByTagName("select")[0] : null;
                var updatedPoste = (sel && sel.value) ? sel.value : personnelData[idx].poste;

                personnelData[idx].nom = updatedNom;
                personnelData[idx].poste = updatedPoste;
            }

            // 3. Nettoyer les données : on supprime les entrées totalement vides pour éviter les lignes fantômes
            var cleanedPersonnel = [];
            for (var k = 0; k < personnelData.length; k++) {
                if (personnelData[k].nom !== "" || personnelData[k].poste !== "") {
                    cleanedPersonnel.push(personnelData[k]);
                }
            }
            personnelData = cleanedPersonnel;

            // 4. On finalise la sauvegarde et on referme l'interface
            SaveData();
            isEditingPlanning = false;
            document.getElementById("btnModifierPlanning").style.display = "inline-block";
            document.getElementById("btnValidateEdit").style.display = "none";
            document.getElementById("btnCancelEdit").style.display = "none";
            RenderMSNTable();
            RenderEquipeVacations();
            EgaliserHauteursPlanningCards();
        }

        function EgaliserHauteursPlanningCards() {
            var cards = document.querySelectorAll("#viewModePlanningEquipes .card");
            if (!cards || cards.length === 0) return;
            var rectTop = cards[0].getBoundingClientRect().top;
            var available = Math.max(150, (document.documentElement.clientHeight || window.innerHeight) - rectTop - 25);
            for (var i = 0; i < cards.length; i++) {
                var card = cards[i];
                var body = card.children[1]; // .planning-card-body (enfant direct)
                var title = card.children[0]; // .card-title (enfant direct)
                var titleH = title ? title.offsetHeight : 0;
                var spacer = body ? body.getElementsByClassName("planning-spacer")[0] : null;
                card.style.height = "";
                if (body) { body.style.height = ""; body.style.overflowY = ""; }
                if (spacer) spacer.style.height = "0";
                var scrolle = false;
                if (isEditingPlanning) {
                    // Édition : toutes les cartes remplissent la même hauteur -> bas alignés,
                    // gap cohérent ; le corps défile si une équipe a plus de membres que la place.
                    if (body) { scrolle = true; }
                } else {
                    // Consultation : chaque carte épouse son propre tableau ; on cape au viewport
                    // (scroll interne) uniquement si le tableau dépasse la hauteur visible.
                    if (card.offsetHeight > available && body) { scrolle = true; }
                }
                if (scrolle) {
                    card.style.height = available + "px";
                    body.style.height = (available - titleH) + "px";
                    body.style.overflowY = "auto";
                    if (spacer) spacer.style.height = "20px"; // gap bas (IE ignore padding-bottom au scroll)
                }
            }
        }

        // --- 7.5 FONCTION GLOBALE DE CADRAGE ET D'ALIGNEMENT DES HAUTEURS ---
        function AjusterHauteursLayout() {
            // 1. Alignement hauteur TOP ROW (Effectif + Guide)
            var eff = document.getElementById("cardEffectifs");
            var aide = document.getElementById("cardAide");
            if (eff && aide) {
                var effContent = eff.querySelector(".card-content");
                var effTitle = eff.querySelector(".card-title");
                // Réinitialisation avant mesure des hauteurs naturelles
                eff.style.height = ""; aide.style.height = "";
                if (effContent) effContent.style.height = "";
                var maxH = Math.max(eff.offsetHeight, aide.offsetHeight);
                eff.style.height = maxH + "px";
                aide.style.height = maxH + "px";
                // La cellule-table remplit la carte ; le centrage vertical est géré par vertical-align:middle
                if (effContent && effTitle) {
                    effContent.style.height = (maxH - effTitle.offsetHeight - 2) + "px";
                }
            }

            // 2. Alignement hauteur BOTTOM ROW (Alertes + Stats)
            var alerte = document.getElementById("cardAlertes");
            var stats = document.getElementById("cardStats");
            if (alerte && stats) {
                var alertesContent = document.getElementById("alertesContent");
                var statsContent = document.getElementById("statsContent");
                var icStatsEmpty = document.getElementById("icStatsEmpty");
                var icAlertesEmpty = document.getElementById("icAlertesEmpty");
                // Réinitialisation avant mesure (le padding-top stale fausse offsetHeight)
                alerte.style.height = ""; stats.style.height = "";
                if (alertesContent) { alertesContent.style.height = ""; alertesContent.style.overflowY = ""; alertesContent.style.paddingBottom = ""; }
                if (statsContent) { statsContent.style.height = ""; statsContent.style.overflowY = ""; statsContent.style.paddingBottom = ""; }
                if (icStatsEmpty) icStatsEmpty.style.paddingTop = "";
                if (icAlertesEmpty) icAlertesEmpty.style.paddingTop = "";
                var rect = alerte.getBoundingClientRect();
                var bottomH = Math.max(150, window.innerHeight - rect.top - 25);
                alerte.style.height = bottomH + "px";
                stats.style.height = bottomH + "px";
                // Défilement identique sur les deux cartes : sur le card-content entier (sous le titre).
                // padding-bottom à 0 + spacer interne (.scroll-spacer) car IE ignore le padding-bottom au défilement.
                if (alertesContent) {
                    var aTop = alertesContent.getBoundingClientRect().top;
                    var aBoxH = Math.max(50, (alerte.getBoundingClientRect().bottom - 1) - aTop);
                    alertesContent.style.height = aBoxH + "px";
                    alertesContent.style.overflowY = "auto";
                    alertesContent.style.paddingBottom = "0";
                    // Centrage vertical de l'état vide (aucun mouvement)
                    if (icAlertesEmpty) {
                        var ptA = Math.floor((aBoxH - 20 - 20 - icAlertesEmpty.offsetHeight) / 2);
                        if (ptA < 0) ptA = 0;
                        icAlertesEmpty.style.paddingTop = ptA + "px";
                    }
                }
                if (statsContent) {
                    var sTop = statsContent.getBoundingClientRect().top;
                    var sBoxH = Math.max(50, (stats.getBoundingClientRect().bottom - 1) - sTop);
                    statsContent.style.height = sBoxH + "px";
                    statsContent.style.overflowY = "auto";
                    statsContent.style.paddingBottom = "0";
                    // Centrage vertical de l'état vide (aucune extraction)
                    if (icStatsEmpty && icStatsEmpty.style.display !== "none") {
                        var pt = Math.floor((sBoxH - 20 - 20 - icStatsEmpty.offsetHeight) / 2);
                        if (pt < 0) pt = 0;
                        icStatsEmpty.style.paddingTop = pt + "px";
                    }
                }
            }
            
            // 2. Alignement et blocage du défilement pour la table de Création ICs
            var containerTableCreation = document.getElementById("containerTableCreation");
            if (containerTableCreation) {
                var rectTopCrea = containerTableCreation.getBoundingClientRect().top;
                var winHCrea = document.documentElement.clientHeight || document.body.clientHeight;
                var availableH = winHCrea - rectTopCrea - 20;
                containerTableCreation.style.height = Math.max(250, availableH) + "px";
            }
            
            // 3. Alignement et blocage du défilement pour la table d'Extraction ICs
            //    La hauteur est calée sur le viewport réel pour que la scroll bar reste visible.
            var divExtractResultsContainer = document.getElementById("divExtractResultsContainer");
            if (divExtractResultsContainer && divExtractResultsContainer.style.display !== "none") {
                var rectTopExt = divExtractResultsContainer.getBoundingClientRect().top;
                var winHExt = document.documentElement.clientHeight || document.body.clientHeight;
                var availableHExt = winHExt - rectTopExt - 20;
                divExtractResultsContainer.style.height = Math.max(150, availableHExt) + "px";
            }
        }
