        Sub WriteLog(msg)
            On Error Resume Next
            Dim fso, logStream, logPath, wsh
            Set wsh = CreateObject("WScript.Shell")
            logPath = wsh.ExpandEnvironmentStrings("%APPDATA%") & "\SAP_Cloture_Log.txt"
            Set fso = CreateObject("Scripting.FileSystemObject")
            Set logStream = fso.OpenTextFile(logPath, 8, True) 
            logStream.WriteLine Now & " | " & msg
            logStream.Close
            Set logStream = Nothing
            Set fso = Nothing
            Set wsh = Nothing
            On Error GoTo 0
        End Sub

        Function GetSAPSession()
            On Error Resume Next
            Set GetSAPSession = Nothing
            Dim SapGuiAuto, appSAP, conn
            Set SapGuiAuto = GetObject("SAPGUI")
            If Err.Number <> 0 Then Exit Function
            Set appSAP = SapGuiAuto.GetScriptingEngine
            If appSAP.Children.Count = 0 Then Exit Function
            Set conn = appSAP.Children(0)
            If conn.Children.Count = 0 Then Exit Function
            Set GetSAPSession = conn.Children(0)
            On Error GoTo 0
        End Function

        Sub PauseVBS(secondes)
            On Error Resume Next
            Dim wshShell
            Set wshShell = CreateObject("WScript.Shell")
            wshShell.Run "cmd /c ping localhost -n " & (secondes + 1) & " > nul", 0, True
            Set wshShell = Nothing
            On Error GoTo 0
        End Sub

        Function AttendreElementSAP(cheminID, maxSecondes)
            Dim tentatives, obj
            tentatives = 0
            AttendreElementSAP = False
            On Error Resume Next
            Do While tentatives < maxSecondes
                Set obj = oSession.FindById(cheminID, False)
                If Not obj Is Nothing Then
                    AttendreElementSAP = True
                    On Error GoTo 0
                    Exit Function
                End If
                Call PauseVBS(1)
                tentatives = tentatives + 1
            Loop
            On Error GoTo 0
        End Function

        Function OuvrirQM02_Dynamique(icNum)
            Dim msgType, msgText, tentatives
            tentatives = 0
            Do While tentatives < 3
                oSession.EndTransaction
                oSession.StartTransaction "QM02"
                If AttendreElementSAP("wnd[0]/usr/ctxtRIWO00-QMNUM", 5) Then
                    oSession.FindById("wnd[0]/usr/ctxtRIWO00-QMNUM").Text = icNum
                    oSession.FindById("wnd[0]").SendVKey 0
                Else
                    OuvrirQM02_Dynamique = "ERREUR: SAP trop lent (Timeout QM02)"
                    Exit Function
                End If
                
                On Error Resume Next
                msgType = ""
                msgText = ""
                msgType = oSession.FindById("wnd[0]/sbar").MessageType
                msgText = oSession.FindById("wnd[0]/sbar").Text
                On Error GoTo 0
                
                If msgType = "E" Then
                    If tentatives < 2 Then 
                        Call PauseVBS(1)
                        tentatives = tentatives + 1 
                    Else 
                        OuvrirQM02_Dynamique = "ERREUR: " & msgText
                        Exit Function
                    End If
                Else
                    OuvrirQM02_Dynamique = "OK"
                    Exit Function
                End If
            Loop
        End Function

        Function ValiderSauvegardeSAP()
            On Error Resume Next
            Dim sbarType, sbarMsg, maxW
            Call PauseVBS(1)
            sbarType = oSession.FindById("wnd[0]/sbar").MessageType
            sbarMsg = oSession.FindById("wnd[0]/sbar").Text
            maxW = 0
            Do While (sbarType = "W" Or sbarType = "I") And maxW < 3
                oSession.FindById("wnd[0]").SendVKey 0
                Call PauseVBS(1)
                sbarType = oSession.FindById("wnd[0]/sbar").MessageType
                sbarMsg = oSession.FindById("wnd[0]/sbar").Text
                maxW = maxW + 1
            Loop
            If sbarType = "E" Then 
                ValiderSauvegardeSAP = "ERREUR SAP: " & sbarMsg 
            Else 
                ValiderSauvegardeSAP = "OK"
            End If
            On Error GoTo 0
        End Function

        Function VBS_ExecuterActionSpecifique(icNum, actionType)
            Dim res
            WriteLog "------------------------------------------"
            WriteLog "TRAITEMENT SPECIFIQUE IC : " & icNum & " | Action: " & actionType
            If oSession Is Nothing Then Set oSession = GetSAPSession()
            If oSession Is Nothing Then
                WriteLog "ERREUR FATALE : Connexion SAP inactive."
                VBS_ExecuterActionSpecifique = "ERREUR: SAP déconnecté."
                Exit Function
            End If
            
            If Trim(icNum) = "" Then 
                VBS_ExecuterActionSpecifique = "ERREUR: Numéro vide"
                Exit Function
            End If
            
            Select Case actionType
                Case "ANNULER"
                    res = R_Annuler(icNum)
                Case "DEBUT"
                    res = R_Demarrer(icNum)
                Case "FIN"
                    res = R_Finir(icNum)
                Case "QUALITE"
                    res = R_Qualite(icNum)
                Case Else
                    res = "ERREUR: Action inconnue"
            End Select
            VBS_ExecuterActionSpecifique = res
        End Function

        Function R_Annuler(icNum)
            Dim ouverture, d, m, y, strToday, h, mn, strTimeStart, strTimeEnd, futureTime
            Dim shell, nodeKeys, iLoop, nKey, nText, btnFound, tbar1, c2, btnInterrompreFound
            Dim popupValider, checkSave

            WriteLog "Action : Annulation"
            On Error Resume Next
            R_Annuler = "ERREUR: Annulation impossible"
            
            ouverture = OuvrirQM02_Dynamique(icNum)
            If ouverture <> "OK" Then 
                R_Annuler = ouverture
                Exit Function
            End If
            
            d = Right("0" & Day(Date), 2)
            m = Right("0" & Month(Date), 2)
            y = Year(Date)
            strToday = d & "." & m & "." & y
            
            futureTime = DateAdd("n", 2, Now)
            h = Right("0" & Hour(futureTime), 2)
            mn = Right("0" & Minute(futureTime), 2)
            strTimeStart = h & ":" & mn
            strTimeEnd = "23:59"
            
            oSession.FindById("wnd[0]/usr/tabsTAB_GROUP_10/tabp10\TAB01").Select
            oSession.FindById(SAP_TAB_ENTETE & "subSUBSCREEN_2:SAPLIQS0:7901/subUSER0001:SAPLXQQM:0402/ctxtVIQMEL-ZZWISH_STDT").Text = strToday
            oSession.FindById(SAP_TAB_ENTETE & "subSUBSCREEN_2:SAPLIQS0:7901/subUSER0001:SAPLXQQM:0402/ctxtVIQMEL-ZZWISH_STTM").Text = strTimeStart
            oSession.FindById(SAP_TAB_ENTETE & "subSUBSCREEN_2:SAPLIQS0:7901/subUSER0001:SAPLXQQM:0402/ctxtVIQMEL-ZZWISH_ENDDT").Text = strToday
            oSession.FindById(SAP_TAB_ENTETE & "subSUBSCREEN_2:SAPLIQS0:7901/subUSER0001:SAPLXQQM:0402/ctxtVIQMEL-ZZWISH_ENDTM").Text = strTimeEnd
            oSession.FindById("wnd[0]").SendVKey 0
            Call PauseVBS(1)
            
            btnFound = False
            Set shell = oSession.FindById("wnd[0]/shellcont/shell", False)
            If Not shell Is Nothing Then
                Set nodeKeys = shell.GetAllNodeKeys()
                For iLoop = 0 To nodeKeys.Count - 1
                    nKey = nodeKeys(iLoop)
                    nText = UCase(shell.GetItemText(nKey, "Column01"))
                    If Trim(nText) = "" Then 
                        nText = UCase(shell.GetNodeTextByKey(nKey))
                    End If
                    If InStr(nText, "ANNUL") > 0 Then 
                        shell.SelectItem nKey, "Column01"
                        shell.ClickLink nKey, "Column01"
                        btnFound = True
                        Exit For
                    End If
                Next
            End If
            
            If Not btnFound Then 
                R_Annuler = "ERREUR: Demande d'annulation introuvable"
                Exit Function
            End If
            Call PauseVBS(1)
            
            btnInterrompreFound = False
            Set tbar1 = oSession.FindById("wnd[0]/tbar[1]", False)
            If Not tbar1 Is Nothing Then
                For c2 = 0 To tbar1.Children.Count - 1
                    If InStr(1, UCase(tbar1.Children(c2).Text), "INTERROMPRE", 1) > 0 Or InStr(1, UCase(tbar1.Children(c2).Tooltip), "INTERROMPRE", 1) > 0 Then 
                        tbar1.Children(c2).Press
                        btnInterrompreFound = True
                        Exit For
                    End If
                Next
            End If
            
            If Not btnInterrompreFound Then 
                R_Annuler = "ERREUR: Bouton 'Interrompre' introuvable"
                Exit Function
            End If
            
            Call PauseVBS(1)
            oSession.FindById("wnd[0]/tbar[0]/btn[11]").Press
            Call PauseVBS(1)
            
            Set popupValider = oSession.FindById("wnd[1]", False)
            If Not popupValider Is Nothing Then 
                oSession.FindById("wnd[1]/tbar[0]/btn[0]").Press
            End If
            
            checkSave = ValiderSauvegardeSAP()
            If checkSave <> "OK" Then 
                R_Annuler = checkSave 
            Else 
                R_Annuler = "OK"
            End If
            On Error GoTo 0
        End Function

        function HandlePaste(e, targetInput) {
            var ev = e || window.event; 
            var pastedText = "";
            
            if (window.clipboardData && window.clipboardData.getData) {
                pastedText = window.clipboardData.getData('Text');
            } else if (ev.clipboardData && ev.clipboardData.getData) {
                pastedText = ev.clipboardData.getData('text/plain');
            }
            
            if (!pastedText) return true; 

            if (ev.preventDefault) ev.preventDefault(); 
            ev.returnValue = false;

            try {
                var rows = pastedText.split('\n'); 
                var td = getClosestTag(targetInput, 'TD');
                var tr = getClosestTag(td, 'TR'); 
                var tbody = tr.parentNode;
                
                var startRowIndex = tr.rowIndex - tbody.rows[0].rowIndex; 
                var startColIndex = td.cellIndex;

                for (var i = 0; i < rows.length; i++) {
                    var rowStr = rows[i].replace(/\r/g, ''); 
                    if (rowStr === '' && i === rows.length - 1) continue; 
                    
                    if (startRowIndex + i >= tbody.rows.length) {
                        AjouterLigneSaisieIC();
                    }
                    
                    var currentRow = tbody.rows[startRowIndex + i]; 
                    var cells = rowStr.split('\t');
                    
                    for (var j = 0; j < cells.length; j++) {
                        if (startColIndex + j >= currentRow.cells.length) continue;
                        
                        var currentCell = currentRow.cells[startColIndex + j];
                        var inputEl = currentCell.querySelector('input, select');
                        if (inputEl) {
                            var val = NettoyerChaine(cells[j]);
                            if (inputEl.type === 'checkbox') {
                                var vLow = val.toLowerCase(); 
                                var shouldCheck = (vLow === 'vrai' || vLow === 'true' || vLow === '1' || vLow === 'x' || vLow === 'oui' || vLow === 'yes');
                                inputEl.checked = shouldCheck; 
                                if (shouldCheck) uncheckOthers(inputEl);
                            } else if (inputEl.tagName === 'SELECT') {
                                for(var opt = 0; opt < inputEl.options.length; opt++) {
                                    if(inputEl.options[opt].value.toUpperCase() === val.toUpperCase()) { 
                                        inputEl.selectedIndex = opt; 
                                        break; 
                                    }
                                }
                            } else { 
                                inputEl.value = val; 
                            }
                        }
                    }
                }
            } catch (err) {}
            return false;
        }

        function FormatDateString(dateObj) {
            var dd = dateObj.getDate(); var mm = dateObj.getMonth() + 1; var yyyy = dateObj.getFullYear();

        function parseDateString(dateStr) {
            if (!dateStr) return null;
            var cleanStr = dateStr.replace(/[\.-]/g, '/'); 
            var parts = cleanStr.split('/'); if (parts.length !== 3) return null;
            var d, m, y;
            if (parts[0].length === 4) {
                y = parseInt(parts[0], 10); m = parseInt(parts[1], 10) - 1; d = parseInt(parts[2], 10);
            } else {
                d = parseInt(parts[0], 10); m = parseInt(parts[1], 10) - 1;
                y = parts[2].length === 4 ? parseInt(parts[2], 10) : parseInt(parts[2], 10) + 2000;
            }
            if (isNaN(d) || isNaN(m) || isNaN(y)) return null;
            return new Date(y, m, d);
        }

        function buildCalendar(month, year) {
            var dp = document.getElementById('custom-datepicker');
            var html = '<div class="dp-header"><button type="button" class="prev" onclick="changeDpMonth(-1, event)">&lt;</button><span>' + moisNoms[month] + ' ' + year + '</span><button type="button" class="next" onclick="changeDpMonth(1, event)">&gt;</button><div style="clear:both;"></div></div><table class="dp-table"><tr><th>Lu</th><th>Ma</th><th>Me</th><th>Je</th><th>Ve</th><th>Sa</th><th>Di</th></tr><tr>';
            var firstDay = new Date(year, month, 1).getDay(); if (firstDay === 0) firstDay = 7; 
            var daysInMonth = new Date(year, month + 1, 0).getDate(); var d = 1;
            for (var i = 1; i < firstDay; i++) { html += '<td></td>'; }
            for (var i = firstDay; i <= 7; i++) { html += '<td onclick="pickDpDate('+d+','+month+','+year+')">'+d+'</td>'; d++; }
            html += '</tr>';
            while (d <= daysInMonth) {
                html += '<tr>';
                for (var i = 1; i <= 7; i++) {
                    if (d <= daysInMonth) { html += '<td onclick="pickDpDate('+d+','+month+','+year+')">'+d+'</td>'; d++; } else { html += '<td></td>'; }
                }
                html += '</tr>';
            }
            dp.innerHTML = html + '</table>';
        }

        function showCalendar(input, event) {
            var ev = event || window.event;
            if(ev) { 
                if(ev.stopPropagation) ev.stopPropagation(); else ev.cancelBubble = true;
                if(ev.preventDefault) ev.preventDefault(); else ev.returnValue = false;
            }
            dpInput = input;
            if(input.value) {
                var parts = input.value.split('/');
                if(parts.length === 3) {
                    var y = parts[2].length === 4 ? parseInt(parts[2], 10) : parseInt(parts[2], 10) + 2000;
                    var m = parseInt(parts[1], 10) - 1; var d = parseInt(parts[0], 10);
                    if(!isNaN(y) && !isNaN(m) && !isNaN(d)) dpDate = new Date(y, m, d);
                }
            } else { dpDate = new Date(); }
            var rect = input.getBoundingClientRect(); var dp = document.getElementById('custom-datepicker');
            dp.style.left = rect.left + 'px'; dp.style.top = (rect.bottom + 2) + 'px'; dp.style.display = 'block';
            buildCalendar(dpDate.getMonth(), dpDate.getFullYear());
        }

        document.onpaste = function(e) {
            var ev = e || window.event; 
            if (!ev) return true;
            var target = ev.target || ev.srcElement;
            var table = getClosestTag(target, 'TABLE');
            
            if (!table || table.id !== 'tableSaisieCreation') return true; 

            var pastedText = "";
            if (window.clipboardData && window.clipboardData.getData) {
                pastedText = window.clipboardData.getData('Text');
            } else if (ev.clipboardData && ev.clipboardData.getData) {
                pastedText = ev.clipboardData.getData('text/plain');
            }
            if (!pastedText) return true;

            if(ev.preventDefault) ev.preventDefault(); else ev.returnValue = false;

            try {
                var rows = pastedText.split('\n'); 
                var tr = getClosestTag(target, 'TR'); 
                var tbodyInner = tr.parentNode;
                var startRowIndex = -1, startColIndex = -1;
                for(var r=0; r<tbodyInner.children.length; r++) { if(tbodyInner.children[r]===tr) { startRowIndex=r; break; } }
                var td = getClosestTag(target, 'TD');
                for(var c=0; c<tr.children.length; c++) { if(tr.children[c]===td) { startColIndex=c; break; } }

                for (var i = 0; i < rows.length; i++) {
                    var rowStr = rows[i].replace(/\r/g, ''); 
                    if (rowStr === '' && i === rows.length - 1) continue;
                    if (startRowIndex + i >= tbodyInner.children.length) AjouterLigneSaisieIC();
                    var currentRow = tbodyInner.children[startRowIndex + i]; var cells = rowStr.split('\t');
                    for (var j = 0; j < cells.length; j++) {
                        if (startColIndex + j >= currentRow.children.length) continue;
                        var inputEl = currentRow.children[startColIndex + j].querySelector('input, select');
                        if (inputEl) {
                            var val = NettoyerChaine(cells[j]);
                            if (inputEl.type === 'checkbox') {
                                var vLow = val.toLowerCase(); var shouldCheck = (vLow === 'vrai' || vLow === 'true' || vLow === '1' || vLow === 'x' || vLow === 'oui' || vLow === 'yes');
                                inputEl.checked = shouldCheck; if(shouldCheck) uncheckOthers(inputEl);
                            } else if (inputEl.tagName === 'SELECT') {
                                for(var opt = 0; opt < inputEl.options.length; opt++) {
                                    if(inputEl.options[opt].value.toUpperCase() === val.toUpperCase()) { inputEl.selectedIndex = opt; break; }
                                }
                            } else { inputEl.value = val; }
                        }
                    }
                }
            } catch (err) {
                // Erreur silencieuse de collage pour ne pas bloquer l'IHM
            }
            return false;
        };
